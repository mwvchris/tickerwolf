Route::get('/tickers/search', function (Request $request) {
    $q = trim($request->get('q', ''));

    if ($q === '' || strlen($q) < 1) {
        return response()->json([]);
    }

    return Ticker::query()
        ->where('ticker', 'like', "%{$q}%")
        ->orWhere('name', 'like', "%{$q}%")
        ->limit(10)
        ->get(['id', 'ticker', 'name']);
});
